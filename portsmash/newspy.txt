#include "spy.h"

# args   : rdi (buf) rsi rdx rcx r8 r9
# scratch: rax r10 r11

.text

.global x64_portsmash_spy
.p2align 4
x64_portsmash_spy:
# just some non-degenerate values
add %rdi, %r8
add %rdi, %r9
add %rdi, %r10
add %rdi, %r11

mov $SPY_NUM_TIMINGS, %rcx

1:
lfence
rdtsc # rdx:rax
lfence
mov %rax, %rsi

/* move-elimination / rename-stage probe */
.rept 128
mov %r8,  %r9
mov %r9,  %r10
mov %r10, %r11
mov %r11, %r8
.endr

lfence
rdtsc
shl $32, %rax
or %rsi, %rax
mov %rax, (%rdi)
add $8, %rdi
dec %rcx
jnz 1b

ret
